				id = chinese_revolt.2

			}
		}
		}	
		add_prestige = -75
	}
	
	option = {
		name = chinese_revolt.1.b #同意
		random_held_title = {
			limit = {

			}
			county = { remove_county_modifier = county_corruption_unreturned_soliders_modifier }
		}
		remove_short_term_gold = major_gold_value
	
		add_prestige = 25
	}
}

chinese_revolt.2 = { #戍卒兵变
	type = character_event
	title = chinese_revolt.2.t
	desc = chinese_revolt.2.desc
	theme = war
	left_portrait = root
    
	
	immediate = {
		random_held_title = {
			limit = { 
			tier = tier_county
			county = { has_county_modifier = county_corruption_unreturned_soliders_modifier } }
			save_scope_as = peasant_county
			remove_county_modifier = county_corruption_unreturned_soliders_modifier
		}
		create_character = {
			location = scope:peasant_county.title_province
			template = peasant_faction_leader_template
			faith = scope:peasant_county.faith
			culture = scope:peasant_county.culture
			gender_female_chance = {
				if = {
					limit = { scope:peasant_county.faith = { has_doctrine = doctrine_gender_male_dominated } }
					add = 0
				}
				else_if = {
					limit = { scope:peasant_county.faith = { has_doctrine = doctrine_gender_female_dominated } }
					add = 100
				}
				else = {
					add = 50
				}
			}
			save_scope_as = peasant_leader
			
		}
	
	create_dynamic_title = {
		tier = empire
		name = FACTION_chinese_peasant_title_NAME
	}
	create_title_and_vassal_change = {
		type = created
		save_scope_as = change
		add_claim_on_loss = no
	}

	scope:new_title = {
		set_capital_county = scope:peasant_county
		set_landless_title = yes
		set_destroy_on_succession = yes
		set_delete_on_destroy = yes
		set_no_automatic_claims = yes
		set_definitive_form = yes
		set_can_be_named_after_dynasty = no
		change_title_holder = {
			holder = scope:peasant_leader
			change = scope:change
		}		
		set_variable = {
			name = chinese_peasant_title
			value = 1
		}
	}
	scope:peasant_leader = {
		random_list = {
			10 = { add_prestige = monumental_prestige_gain }
			20 = { add_prestige = massive_prestige_gain }
			30 = { add_prestige = major_prestige_gain }
			40 = { add_prestige = medium_prestige_gain }
		}
	}
	resolve_title_and_vassal_change = scope:change
	scope:new_title = { generate_coa = factions }
	scope:peasant_leader = { 
	start_war = { 
	cb = solider_war 
	target = ROOT
	target_title = ROOT.primary_title
	}
	 
     spawn_army = {
		uses_supply = no
		inheritable = no
		name = shuzu_event_troops
		men_at_arms = {
			type = light_footmen
			stacks = 8
		}
		men_at_arms = {
			type = bowmen
			stacks = 2
		}
		men_at_arms = {
			type = pikemen_unit
			stacks = 4
		}
		men_at_arms = {
			type = light_horsemen
			stacks = 1
		}
		location = scope:peasant_county.title_province
	  }
	}

	
	}
	
	option = {
		name = chinese_revolt.2.a #不允许
		
	}
	
}

chinese_revolt.3 = {
	hidden = yes

	trigger = {
		has_trait = peasant_leader
		exists = scope:war
		#OR = {
		#	scope:county.culture = root.culture
		#	scope:county.religion = root.religion
		#}
	}

	immediate = {
		# Determine how many rebel reinforcements we need to spawn, based on the total number of provinces captured.
        if = {
		 limit = {
		  scope:county = {
		   NOT = { has_county_modifier = county_corruption_unreturned_soliders_modifier }
		   }
		   }
		 # Spawn the army on the captured province.
		 spawn_army = {
			name = event_troop_default
			levies = {
				add = prestige
				multiply = 0.25
			}
			location = scope:barony.title_province
		  }
		 # Spawn the army on the captured province.
		 spawn_army = {
			name = event_troop_default
			levies = {
				add = prestige
				multiply = 0.75
			}
			location = scope:county.title_province
		  }
		}
        if = {
		 limit = {
		  scope:county = {
		   has_county_modifier = county_corruption_unreturned_soliders_modifier
		   }
		  }
		 # Spawn the army on the captured province.
		 spawn_army = {
		  uses_supply = no
		  inheritable = no
		  name = shuzu_event_troops
		  men_at_arms = {
			type = light_footmen
			stacks = 4
		  }
		  men_at_arms = {
			type = bowmen
			stacks = 2
		  }
		  men_at_arms = {
			type = pikemen_unit
			stacks = 3
		  }
		  men_at_arms = {
			type = light_horsemen
			stacks = 1
		  }
		  location = scope:barony.title_province
			war = scope:war
	      }
		 spawn_army = {
		  uses_supply = no
		  inheritable = no
		  name = shuzu_event_troops
		  men_at_arms = {
			type = light_footmen
			stacks = 4
		  }
		  men_at_arms = {
			type = bowmen
			stacks = 2
		  }
		  men_at_arms = {
			type = pikemen_unit
			stacks = 3
		  }
		  men_at_arms = {
			type = light_horsemen
			stacks = 1
		  }
		  location = scope:county.title_province
			war = scope:war
	      }
		  remove_county_modifier = county_corruption_unreturned_soliders_modifier
		}
		# Additional loss of county control, can be up to -50 for large counties.
		every_in_list = {
			list = occupied_baronies
			scope:county = {
				change_county_control = -10
				remove_county_modifier = county_corruption_hengzhengbaolian_modifier
			}
		}
	}

	after = {
		remove_variable = number_rebel_reinforcements
	}
}
